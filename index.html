const {Composite, ImageView, TextView, ScrollView, ui} = require('tabris'); const IMAGE_PATH = 'https://bettydraws/TabrisPortfolio/images/'; const INITIAL_TITLE_COMPOSITE_OPACITY = 0.85; const TITLE = 'BETTYDRAWS'; const SUBTITLE = 'Elizabeth Saunders Portfolio'; const BIO = 'Hello, my name is Elizabeth Saunders, a student currently studying at Confederation College in their IMD (Interactive Media Developement) program, studying towards an Ontario College Advanced Diploma. A resident of Thunder Bay, Ontario with a passion for the Arts and a few opportunities to have my artwork up in the Thunder Bay Art Gallery. As well as a few opportunities to compete in the Skills Ontario Competition for Architectural Technology and Design, and the Graphic Design Studio Production Scopes. On my portfolio on bettydraws.com you are most likely to find my love of graphic design, photography, 3D design, drawing, and so on. A few images featured on my portfolio have also been on Pixabay.'; let titleCompY = 0; ui.statusBar.background = rgba(150, 10, 0, 1); let scrollView = new ScrollView({ left: 0, right: 0, top: 0, bottom: 0 }).appendTo(ui.contentView); let imageView = new ImageView({ left: 0, top: 0, right: 0, image: 'images/bettydraws.jpg', scaleMode: 'fill' }).appendTo(scrollView); let contentComposite = new Composite({ left: 0, right: 0, top: '#titleComposite', height: 1000, background: 'white' }).appendTo(scrollView); new TextView({ left: 20, right: 20, top: 30, text: BIO }).appendTo(contentComposite); let titleComposite = new Composite({ left: 0, right: 0, height: 78, id: 'titleComposite', background: rgba(150, 10, 0, INITIAL_TITLE_COMPOSITE_OPACITY) }).appendTo(scrollView); new TextView({ left: 20, top: 16, right: 20, text: SUBTITLE, font: 'bold 16px', textColor: 'white' }).appendTo(titleComposite); new TextView({ left: 20, bottom: 18, right: 16, top: 'prev()', text: TITLE, font: 'bold 24px', textColor: 'white' }).appendTo(titleComposite); scrollView.on('resize', ({height}) => { imageView.height = height / 2; let titleCompHeight = titleComposite.height; titleCompY = Math.min(imageView.height - titleCompHeight, height / 2); titleComposite.top = titleCompY; }); scrollView.on('scrollY', ({offset}) => { imageView.transform = {translationY: Math.max(0, offset * 0.4)}; titleComposite.transform = {translationY: Math.max(0, offset - titleCompY)}; let opacity = calculateTitleCompositeOpacity(offset, titleCompY); titleComposite.background = rgba(150, 10, 0, opacity); }); function calculateTitleCompositeOpacity(scrollViewOffsetY, titleCompY) { let titleCompDistanceToTop = titleCompY - scrollViewOffsetY; let opacity = 1 - (titleCompDistanceToTop * (1 - INITIAL_TITLE_COMPOSITE_OPACITY)) / titleCompY; return opacity <= 1 ? opacity : 1; } function rgba(r, g, b, a) { return 'rgba(' + r + ',' + g + ',' + b + ',' + a + ')'; }
